#validates the arguments
function parameterChecker(){
    firstName=$1

        if  ! [[ "$firstName" =~ ^[0-9]+$ ]]
        then

            then

                echo "Second argument must be number"
            fi

            echo "The first argument should not contain symbols and numbers"
        fi
    else

        echo
        echo "Wrong argument provided"
        echo 
        echo "Provide first argument as First Name"
        echo
        echo "Provide second argument as LondonID"
        echo 
        echo "====================================="
        exit
    fi
}
#validates secretKey
function secretKey(){
  i=0
  key=6441
  while [ $i -lt 4 ] 
  do
    echo -e "Enter the secretkey: \c"
    read secretkey
    if [ $i -le 3 ] 
    then 
      #checks whether the secretkey is integers or not
      if ! [[ "$secretkey" =~ ^[0-9]+$ ]]
      then
          echo "--------------------------"
          echo
          echo "Sorry integers only"
          echo 
          echo "Please enter the key again"
          echo
          echo "--------------------------"
          ((i++))
      else
        #checks whether the secretkey is equal to key or not
        if [ $secretkey -eq $key ]
        then
            #makes a call to the function
            nameAndId
            break
        else 
            echo "--------------------------"
            sleep 1
            echo
            echo "You entered the wrong key"
            sleep 1
            echo
            echo "Please enter the key again"
            echo 
            echo "------------------------------"
            ((i++))
        fi
      fi 
    fi
    if [ $i -eq 4 ]
    then 
      echo "You entered wrong key four times"
      echo "Please run the program again and provide your secretKey"
    fi 
  done
}

function continue(){
